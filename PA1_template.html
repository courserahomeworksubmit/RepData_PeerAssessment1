An auxilliary function and its vectorized version~

h1 <- function ( d )
{
  a <- weekdays ( as.Date ( d ) , abbreviate = TRUE )
  if ( a == "Mon" | a == "Tue" | a == "Wed" | a == "Thu" | a == "Fri" )
  {
    return ( "weekday" )
  }

  return ("weekend")
}

h <- Vectorize ( h1 )
